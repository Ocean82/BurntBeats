// backend/melody-generator.ts
import { music21 } from 'music21';

export class MelodyGenerator {
  async generateMelody(genre: string, mood: string, tempo: number): Promise<any> {
    // Use music21 or another library to generate melody
    const melody = music21.melody.generate({
      genre,
      mood,
      tempo,
      duration: 120, // Example duration in seconds
    });

    return melody;
  }
}
